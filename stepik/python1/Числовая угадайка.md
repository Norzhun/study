import random  
  
print('Добро пожаловать в числовую угадайку!:)')  
print('Укажите правую границу для угадываемого числа', 'начинаем!', sep='\n')  
right = int(input())  # Правая граница  
  
  
def is_valid(s):  # Проверяем число ли это, и входит ли в границы  
    return s.isdigit() and 1 <= int(s) <= right  
  
  
def next_game():  # Возможность следующей игры  
    question = input('Хотите продолжить ("д"/"н")?\n')  
    if question in 'д, l':  
        print('Продолжаем, вводите число!')  
        return True  
    elif question in 'н, y':  
        print('До новых встреч!')  
        return False  
    else:  
        print('Нажмите либо на "н", либо на "д"')  
  
  
def game():  
    num = random.randint(1, right)  # Рандомное число от 1 до указанной правой границы (включительно).  
    count = 0  # Счетчик  
    while True:  
        n = input()  
        if is_valid(n):  
            n = int(n)  
            if n < num:  
                count += 1  
                print('Ваше число меньше загаданного, попробуйте ввести большее число')  
            elif n > num:  
                count += 1  
                print('Ваше число больше загаданного, попробуйте ввести меньшее число')  
            else:  
                count += 1  
                print('Вы угадали за', count, 'попыток,', 'поздравляем!')  
                if next_game():  # Проверка на следующую игру  
                    count = 0  
                    num = random.randint(1, right)  
                else:  
                    break  
        else:  
            print('А может быть все-таки введем целое число от 1 до', right, '?')  
  
  
game()
